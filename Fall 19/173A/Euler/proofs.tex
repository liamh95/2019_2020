\section*{3.3}
Let $p$ and $q$ be distinct primes and let $e$ and $d$ be positive integers satisfying
\[
de \equiv 1 \pmod{(p-1)(q-1)}.
\]
Suppose further that $c$ is an integer with $\gcd(c, pq)>1$. Prove that
\[
x \equiv c^d\pmod{pq}
\]
is a solution to the congruence $x^e \equiv c\pmod{pq}$.
\begin{proof}
	Suppose $c$ is divisible by both $p$ and $q$. Then $c \equiv 0\pmod{pq}$ 
\end{proof}
